490  void onMessageReceivedImpl() {
491 
492  Command cmd = lastCMD;
493  DeviceConnection *conn = deviceConnection;
494 
495  messageReceived = false;
496  connected = true;
497  keepAliveTime = millis();
498  keepAliveMiss = 0;
499 
500  bool cont = true; // TODO: Chama handlers(functions), se retornar false abota a continuacao;
501 
502  debug("CType:", cmd.type);
503  // showFreeRam();
504 
505  // Directed to a device (Like On/OFF or more complex)
506  if (cmd.deviceID > 0) {
507  Device *foundDevice = getDevice(cmd.deviceID);
508  if (foundDevice != NULL) {
509  debugChange(foundDevice->id, cmd.value);
510  foundDevice->setValue(cmd.value, false);
511  foundDevice->deserializeExtraData(&cmd, conn);
512  notifyReceived(ResponseStatus::SUCCESS);
513  } else {
514  notifyReceived(ResponseStatus::NOT_FOUND);
515  }
516  // User-defined command, this is an easy way to extend OpenDevice protocol.
517  } else if (cmd.type == CommandType::USER_COMMAND) {
518  String name = conn->readString();
519  for (int i = 0; i < commandsLength; i++) {
520  // if(debugMode){ debug("Call function:"); debug(name); }
521  if (name.equals(commands[i].command)) {
522  notifyReceived(ResponseStatus::SUCCESS);
523  (*commands[i].function)();
524  }
525  }
526  } else if (cmd.type == CommandType::PING_REQUEST) {
527 
528  send(resp(CommandType::PING_RESPONSE, 0, ResponseStatus::SUCCESS));
529 
530  } else if (cmd.type == CommandType::RESET) {
531 
532  reset();
533 
534  // Send response: GET_DEVICES_RESPONSE;ID;Index;Length;[ID, PIN, VALUE, TARGET, SENSOR?, TYPE];
535  // NOTE: This message is sent to each device
536  } else if (cmd.type == CommandType::GET_DEVICES) {
537 
538  char buffer[DATA_BUFFER] = {0}; // FIXME: daria para usar o mesmo buffer do deviceConnection ??
539 
540  for (int i = 0; i < deviceLength; ++i) {
541  Device *device = getDeviceAt(i);
542 
543  conn->doStart();
544  conn->print(CommandType::GET_DEVICES_RESPONSE);
545  conn->doToken();
546  conn->print(cmd.id);
547  conn->doToken();
548  conn->print(i+1);
549  conn->doToken();
550  conn->print(deviceLength);
551  conn->doToken();
552 
553  device->toString(buffer);
554 
555  conn->print(buffer); // Write array to connection..
556 
557  memset(buffer, 0, sizeof(buffer));
558 
559  conn->doEnd();
560  }
561 
562  // Save devices ID on storage
563  } else if (cmd.type == CommandType::SYNC_DEVICES_ID) {
564 
565  // conn->printBuffer();
566 
567  int length = conn->readInt();
568 
569  Config.devicesLength = length;
570 
571  LOG_DEBUG("SYNC", length);
572 
573  if(length!= deviceLength) { // Invalid Match
574  notifyReceived(ResponseStatus::BAD_REQUEST);
575  return;
576  }
577 
578  for (size_t i = 0; i < length; i++) {
579  int uid = conn->readInt();
580  if(uid > MAX_DEVICE_ID){
581  LOG_DEBUG_S("MAX_ID ERROR");
582  notifyReceived(ResponseStatus::BAD_REQUEST);
583  return;
584  }
585 
586  devices[i]->id = uid;
587  Config.devices[i] = uid;
588  // Serial.print("SYNC :: ");Serial.print(i);Serial.print(" => ");Serial.println(devices[i]->id, DEC);
589  }
590 
591  save();
592  notifyReceived(ResponseStatus::SUCCESS);
593 
594 
595  } else if (cmd.type == CommandType::FIRMWARE_UPDATE) {
596 
597  #ifdef ESP8266HTTPUPDATE_H_
598  String url = conn->readString();
599  RemoteUpdate.updateFromURL(url);
600  #endif
601 
602  }else{
603 
604  // TODO: Send response: UNKNOW_COMMAND
605 
606  }
607 
608  }
